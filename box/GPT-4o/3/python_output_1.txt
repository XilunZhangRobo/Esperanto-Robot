
import numpy as np  # Import numpy for any necessary calculations

# Step 1: Grasp CubeA
open_gripper()
cubeA_graspable_point = get_graspable_point('cubeA')
move_to_position(np.array([cubeA_graspable_point[0], cubeA_graspable_point[1], cubeA_graspable_point[2] + 0.1]))  # Move above CubeA
move_to_position(cubeA_graspable_point)  # Move down to grasp CubeA
close_gripper()

# Step 2: Place CubeA on CubeB
cubeB_size = get_size('cubeB')
cubeB_top_position = np.array([cubeB_size[0] / 2, cubeB_size[1] / 2, cubeB_size[2]])  # Assuming CubeB is at origin
move_to_position(np.array([cubeB_top_position[0], cubeB_top_position[1], cubeB_top_position[2] + 0.1]))  # Move above CubeB
move_to_position(cubeB_top_position)  # Move down to place CubeA on CubeB
open_gripper()

# Step 3: Grasp BallA
open_gripper()
ballA_graspable_point = get_graspable_point('ballA')
move_to_position(np.array([ballA_graspable_point[0], ballA_graspable_point[1], ballA_graspable_point[2] + 0.1]))  # Move above BallA
move_to_position(ballA_graspable_point)  # Move down to grasp BallA
close_gripper()

# Step 4: Place BallA on CubeA
cubeA_size = get_size('cubeA')
cubeA_top_position = np.array([cubeB_top_position[0], cubeB_top_position[1], cubeB_top_position[2] + cubeA_size[2]])  # CubeA is on top of CubeB
move_to_position(np.array([cubeA_top_position[0], cubeA_top_position[1], cubeA_top_position[2] + 0.1]))  # Move above CubeA
move_to_position(cubeA_top_position)  # Move down to place BallA on CubeA
open_gripper()
