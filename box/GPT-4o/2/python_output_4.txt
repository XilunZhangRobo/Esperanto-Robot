
import numpy as np  # Import numpy for any potential calculations

# Step 1: Open the gripper
open_gripper()

# Step 2: Get the graspable point of cubeA
cubeA_graspable_point = get_graspable_point('cubeA')

# Step 3: Move above cubeA
move_to_position(np.array([cubeA_graspable_point[0], cubeA_graspable_point[1], cubeA_graspable_point[2] + 0.1]))  # Move 10 cm above the graspable point

# Step 4: Move down to grasp cubeA
move_to_position(cubeA_graspable_point)

# Step 5: Close the gripper to grasp cubeA
close_gripper()

# Step 6: Lift cubeA
move_to_position(np.array([cubeA_graspable_point[0], cubeA_graspable_point[1], cubeA_graspable_point[2] + 0.1]))  # Lift 10 cm up

# Step 7: Get the top position of cubeB
cubeB_size = get_size('cubeB')
cubeB_top_position = get_graspable_point('cubeB') + np.array([0, 0, cubeB_size[2]])  # Top position of cubeB

# Step 8: Move above cubeB
move_to_position(np.array([cubeB_top_position[0], cubeB_top_position[1], cubeB_top_position[2] + 0.1]))  # Move 10 cm above the top of cubeB

# Step 9: Move down to place cubeA on cubeB
move_to_position(cubeB_top_position)

# Step 10: Open the gripper to release cubeA
open_gripper()

# Step 11: Move up to clear the area
move_to_position(np.array([cubeB_top_position[0], cubeB_top_position[1], cubeB_top_position[2] + 0.1]))  # Move 10 cm up
