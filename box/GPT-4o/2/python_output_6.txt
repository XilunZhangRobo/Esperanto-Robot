
import numpy as np  # Import numpy for any necessary calculations

# Step 1: Open the gripper
open_gripper()

# Step 2: Get the graspable point of cubeA
cubeA_graspable_point = get_graspable_point('cubeA')

# Step 3: Move the gripper above cubeA
above_cubeA = np.array(cubeA_graspable_point) + np.array([0, 0, 0.1])  # Adding 0.1 to the z-coordinate to move above
move_to_position(above_cubeA)

# Step 4: Move the gripper down to the graspable point of cubeA
move_to_position(cubeA_graspable_point)

# Step 5: Close the gripper to grasp cubeA
close_gripper()

# Step 6: Get the size of cubeB
cubeB_size = get_size('cubeB')

# Step 7: Calculate the position above cubeB where cubeA should be placed
cubeB_graspable_point = get_graspable_point('cubeB')
above_cubeB = np.array(cubeB_graspable_point) + np.array([0, 0, cubeB_size[2]])

# Step 8: Move the gripper above the calculated position
move_to_position(above_cubeB + np.array([0, 0, 0.1]))  # Adding 0.1 to the z-coordinate to move above

# Step 9: Move the gripper down to the calculated position
move_to_position(above_cubeB)

# Step 10: Open the gripper to release cubeA
open_gripper()
