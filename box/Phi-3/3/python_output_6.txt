
import numpy as np

# Open the gripper
open_gripper()

# Get the graspable point of cubeA
cubeA_graspable_point = get_graspable_point('cubeA')

# Move to the topmost graspable point on cubeA
move_to_position(cubeA_graspable_point)

# Close the gripper to pick up cubeA
close_gripper()

# Get the graspable point of ballA
ballA_graspable_point = get_graspable_point('ballA')

# Move to the topmost graspable point on ballA
move_to_position(ballA_graspable_point)

# Close the gripper to pick up ballA
close_gripper()

# Move to the position where cubeA and ballA need to be stacked on cubeB
move_to_position(get_position_to_stack('cubeB'))

# Close the gripper to release cubeA and ballA onto cubeB
close_gripper()
