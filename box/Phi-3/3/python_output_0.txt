
import numpy as np

# Open the gripper
open_gripper()

# Get the graspable point of cubeA
cubeA_graspable_point = get_graspable_point('cubeA')

# Move to the top-most graspable point on cubeA
move_to_position(cubeA_graspable_point)

# Close the gripper to secure cubeA
close_gripper()

# Get the graspable point of ballA
ballA_graspable_point = get_graspable_point('ballA')

# Move to the top-most graspable point on ballA
move_to_position(ballA_graspable_point)

# Close the gripper to secure ballA
close_gripper()

# Get the graspable point of cubeB
cubeB_graspable_point = get_graspable_point('cubeB')

# Move to the position where cubeA and ballA will be stacked on cubeB
move_to_position(cubeB_graspable_point)

# Open the gripper to release cubeA and ballA onto cubeB
open_gripper()
