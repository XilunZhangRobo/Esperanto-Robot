
import numpy as np

# Step 1: Get the graspable point of cubeA
cubeA_graspable_point = get_graspable_point('cubeA')

# Step 2: Get the size of cubeA
cubeA_size = get_size('cubeA')

# Step 3: Move to the graspable point of cubeA
move_to_position(cubeA_graspable_point)

# Step 4: Open the gripper
open_gripper()

# Step 5: Move to the top of cubeB
cubeB_top_position = get_graspable_point('cubeB') + cubeA_size
move_to_position(cubeB_top_position)

# Step 6: Close the gripper to grasp cubeA
close_gripper()

# Step 7: Move to the desired position on top of cubeB
final_position = get_graspable_point('cubeB') + cubeA_size
move_to_position(final_position)

# Step 8: Close the gripper to secure cubeA on top of cubeB
close_gripper()
