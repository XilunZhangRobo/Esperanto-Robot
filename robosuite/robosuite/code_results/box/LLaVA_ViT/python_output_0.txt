
import numpy as np

# Get the workspace range
x_min, y_min, z_min, x_max, y_max, z_max = get_workspace_range()

# Open the gripper
open_gripper()

# Get the graspable point of cubeA
cubeA_graspable_point = get_graspable_point('cubeA')

# Move to the graspable point of cubeA
move_to_position(cubeA_graspable_point)

# Get the graspable point of cubeB
cubeB_graspable_point = get_graspable_point('cubeB')

# Move to the graspable point of cubeB
move_to_position(cubeB_graspable_point)

# Pick up cubeA
open_gripper()

# Move to the second cube
move_to_position(cubeB_graspable_point)

# Pick up cubeB
open_gripper()

# Move to the first cube
move_to_position(cubeA_graspable_point)

# Stack the two cubes on top of each other
close_gripper()

# Move to the table
move_to_position(np.array([x_min, y_min, z_min]))

# Close the gripper
close_gripper()
